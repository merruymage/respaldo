import easyocr
import numpy as np # Necesario para los tipos np.int32 y np.float64

# Inicializa el lector
reader = easyocr.Reader(['es']) # Asegúrate de usar los idiomas correctos

# Carga la imagen (asegúrate de que la ruta sea correcta)
image_path = 'C:/prueba/27084609001.jpg'
result = reader.readtext(image_path)

# Listas para almacenar los datos
all_coords = []
all_texts = []
all_confidences = []

# Recorre los resultados y almacena la información
for res in result:
    coord = res[0]
    text = res[1]
    conf = res[2]

    all_coords.append(coord)
    all_texts.append(text)
    all_confidences.append(conf)

    print(f"Texto detectado: '{text}' (Confianza: {conf:.2f})")

print("\n--- Resumen de datos almacenados ---")
print(f"Total de textos detectados: {len(all_texts)}")
